    <script>

        const questions = [

            { text: "I like to work with animals, tools, or machines.", type: "R" },

            { text: "I enjoy solving math or science problems.", type: "I" },

            { text: "I like to do creative activities like art, drama, or crafts.", type: "A" },

            { text: "I enjoy helping people and teaching them new things.", type: "S" },

            { text: "I like to lead people and make business decisions.", type: "E" },

            { text: "I like to work with numbers and keep records in order.", type: "C" },

            { text: "I prefer working outdoors rather than in an office.", type: "R" },

            { text: "I am curious about how things work in nature.", type: "I" },

            { text: "I like to express my ideas through writing or music.", type: "A" },

            { text: "I am interested in community service and volunteering.", type: "S" },

            { text: "I enjoy persuading others or selling products.", type: "E" },

            { text: "I am a very organized and detail-oriented person.", type: "C" }

        ];


        const typeData = {

            R: { name: "Realistic", careers: ["Engineer", "Pilot", "Chef", "Athlete"], growth: ["Develop communication skills", "Practice expressing your feelings"] },

            I: { name: "Investigative", careers: ["Scientist", "Software Developer", "Doctor", "Professor"], growth: ["Apply your theories to real life", "Practice teamwork with others"] },

            A: { name: "Artistic", careers: ["Designer", "Writer", "Musician", "Architect"], growth: ["Build consistent work habits", "Learn to follow organizational rules"] },

            S: { name: "Social", careers: ["Teacher", "Counselor", "Nurse", "Social Worker"], growth: ["Learn to set healthy boundaries", "Enhance your technical/digital skills"] },

            E: { name: "Enterprising", careers: ["Manager", "Entrepreneur", "Lawyer", "Politician"], growth: ["Develop active listening skills", "Pay more attention to small details"] },

            C: { name: "Conventional", careers: ["Accountant", "Data Analyst", "Banker", "Admin"], growth: ["Try to be more flexible with changes", "Practice creative problem solving"] }

        };


        let currentIdx = 0;

        let userName = "";

        let answers = new Array(questions.length).fill(null);


        function startTest() {

            userName = document.getElementById('nameInput').value.trim();

            if (userName.length < 2) return alert("Please enter your name (at least 2 letters).");

            document.getElementById('userNameLabel').textContent = userName;

            document.getElementById('start-page').classList.remove('active');

            document.getElementById('quiz-page').classList.add('active');

            renderQuestion();

        }


        function renderQuestion() {

            const q = questions[currentIdx];

            document.getElementById('qText').textContent = q.text;

            document.getElementById('qCount').textContent = `Question ${currentIdx + 1} / ${questions.length}`;

            document.getElementById('progressFill').style.width = `${((currentIdx + 1) / questions.length) * 100}%`;

            

            const likert = document.getElementById('likert');

            const options = ["Strongly Disagree", "Disagree", "Neutral", "Agree", "Strongly Agree"];

            likert.innerHTML = options.map((opt, i) => `

                <button class="likert-btn ${answers[currentIdx] === (i+1) ? 'selected' : ''}" onclick="selectAnswer(${i+1})">

                    ${opt}

                </button>

            `).join('');

            document.getElementById('nextBtn').disabled = answers[currentIdx] === null;

        }


        function selectAnswer(val) {

            answers[currentIdx] = val;

            renderQuestion();

        }


        function goNext() {

            if (currentIdx < questions.length - 1) {

                currentIdx++;

                renderQuestion();

            } else {

                showResult();

            }

        }


        function goBack() {

            if (currentIdx > 0) {

                currentIdx--;

                renderQuestion();

            }

        }


        function showResult() {

            const scores = { R: 0, I: 0, A: 0, S: 0, E: 0, C: 0 };

            questions.forEach((q, i) => scores[q.type] += answers[i]);

            const topType = Object.keys(scores).reduce((a, b) => scores[a] > scores[b] ? a : b);

            

            const data = typeData[topType];

            document.getElementById('finalName').textContent = userName;

            document.getElementById('topTypeTitle').textContent = data.name;

            document.getElementById('careerTags').innerHTML = data.careers.map(c => `<span class="tag">${c}</span>`).join('');

            document.getElementById('growthList').innerHTML = data.growth.map(g => `<li>${g}</li>`).join('');

            

            document.getElementById('quiz-page').classList.remove('active');

            document.getElementById('result-page').classList.add('active');

        }
